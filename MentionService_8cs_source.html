101  public static async Task<string> ResolveMentions(string message)
102  {

104  LogContext.PushProperty("SourceContext", "MentionService");

108  // \_____/
109  // Match ---------------------'
110  //
111  //Regex example:@123456 blah blub
112  // \_____/
113  // Match ----------'
114  //

122  var resolvedMessage = "";

124 
125  foreach (var substr in splitMessage)
126  {
127  if (substr.StartsWith("@"))

129  logger.Information($"Found mention {substr}");

132  switch (mention.TargetType)

134  case MentionTarget.User:
135  resolvedSubstr = (await UserService.GetUser(mention.TargetId)).DisplayName;
136  break;
137  case MentionTarget.Role:

139  break;
140  case MentionTarget.Channel:
141  resolvedSubstr = (await ChannelService.GetChannel(Convert.ToUInt32(mention.TargetId))).ChannelName;
142  break;
143  case MentionTarget.Message:
144  resolvedSubstr = $"#{mention.Id}";
145  break;
146  default:
147  LogContext.PushProperty("Method","ResolveMentions");
148  LogContext.PushProperty("SourceContext", "MentionService");
149  logger.Information($"Could not resolve mention {mention}");
150 
151  resolvedSubstr = "";

153  }
